集群内部访问外部数据库或者中间件一般采用endpoints与service关联方式映射。

  1. #en配置
  2. [root@k8s-master-01 k8s]# more mysql-endpoints.yaml 
  3. apiVersion: v1
  4. kind: Endpoints
  5. metadata:
  6.   name: mysql-dev
  7.   namespace: default
  8. subsets:
  9.   - addresses:
  10.     - ip: 192.168.71.129 #外部数据库地址
  11.     ports:
  12.     - port: 3306
  13. #service配置
  14. [root@k8s-master-01 k8s]# more mysql-service.yaml 
  15. apiVersion: v1
  16. kind: Service
  17. metadata:
  18.   name: mysql-dev
  19.   namespace: default
  20. spec:
  21.   clusterIP: 10.96.2.128 #固定clusterIP
  22.   ports:
  23.   - port: 3306
  24.     targetPort: 3306
  25.     protocol: TCP
  26. #创建testpod测试，需要在同一ns
  27. [root@k8s-master-01 k8s]# kubectl run test-pod1 -it --image=busybox:1.28 --image-pull-policy=IfNotPresent
  28. #通过svc的clusterIP访问数据库，以下提示说明数据库授权后可以正常连接。
  29. / # telnet 10.96.2.128 3306
  30. GHost '192.168.71.134' is not allowed to connect to this MySQL serverConnection closed by foreign host
